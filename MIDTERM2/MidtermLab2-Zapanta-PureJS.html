import csv import heapq import math import os import tkinter as tk from tkinter import ttk, messagebox # ───────────────────────────────────────── # DATA LOADING # ───────────────────────────────────────── def load_graph(filepath): graph = {} # adjacency list: node -> list of (neighbor, attrs) nodes = set() with open(filepath, newline='', encoding='utf-8-sig') as f: reader = csv.DictReader(f) for row in reader: frm = row['From Node'].strip() to = row['To Node'].strip() dist = float(row['Distance (km)']) time = float(row['Time (mins)']) fuel = float(row['Fuel (Liters)']) nodes.add(frm) nodes.add(to) graph.setdefault(frm, []).append((to, {'distance': dist, 'time': time, 'fuel': fuel})) graph.setdefault(to, []).append((frm, {'distance': dist, 'time': time, 'fuel': fuel})) return graph, sorted(nodes) # ───────────────────────────────────────── # DIJKSTRA # ───────────────────────────────────────── def dijkstra(graph, start, end, weight_key): pq = [(0, start, [start], {'distance': 0, 'time': 0, 'fuel': 0})] visited = {} while pq: cost, node, path, totals = heapq.heappop(pq) if node in visited: continue visited[node] = cost if node == end: return cost, path, totals for neighbor, attrs in graph.get(node, []): if neighbor not in visited: new_cost = cost + attrs[weight_key] new_totals = { 'distance': totals['distance'] + attrs['distance'], 'time': totals['time'] + attrs['time'], 'fuel': totals['fuel'] + attrs['fuel'] } heapq.heappush( pq, (new_cost, neighbor, path + [neighbor], new_totals) ) return None, [], {} # ───────────────────────────────────────── # NEW COLOR SCHEME & DIMENSIONS (Compact) # ───────────────────────────────────────── # Blue & Orange theme BG_COLOR = '#0f172a' # Slate 900 CARD_COLOR = '#1e293b' # Slate 800 ACCENT_BLUE = '#3b82f6' # Blue 500 ACCENT_ORANGE = '#f97316' # Orange 500 TEXT_PRIMARY = '#f8fafc' # Slate 50 TEXT_SECONDARY = '#cbd5e1' # Slate 200 EDGE_COLOR = '#475569' # Slate 600 NODE_FILL = '#334155' # Slate 700 HIGHLIGHT_EDGE = '#f97316' # Orange HIGHLIGHT_NODE = '#3b82f6' # Blue # Compact dimensions CANVAS_W = 580 CANVAS_H = 260 RADIUS = 18 WINDOW_W = 620 WINDOW_H = 680 NODE_POSITIONS = { 'NOVELETA': (0.25, 0.35), 'IMUS': (0.42, 0.12), 'BACOOR': (0.75, 0.20), 'KAWIT': (0.25, 0.55), 'DASMA': (0.85, 0.55), 'INDANG': (0.18, 0.85), 'SILANG': (0.52, 0.85), 'GENTRI': (0.85, 0.85), } def draw_map(canvas, graph, nodes, highlight_path=None, animate=False): canvas.delete('all') W, H = CANVAS_W, CANVAS_H # Subtle grid for x in range(0, W, 25): for y in range(0, H, 25): canvas.create_oval(x, y, x+1, y+1, fill=CARD_COLOR, outline='') def pos(name): px, py = NODE_POSITIONS.get(name, (0.5, 0.5)) return int(px * W), int(py * H) positions = {name: pos(name) for name in nodes} drawn_edges = set() hl_edges = set() if highlight_path: for i in range(len(highlight_path) - 1): a, b = highlight_path[i], highlight_path[i + 1] hl_edges.add((min(a, b), max(a, b))) # Edges for node in graph: for neighbor, attrs in graph[node]: key = (min(node, neighbor), max(node, neighbor)) if key in drawn_edges: continue drawn_edges.add(key) x1, y1 = positions[node] x2, y2 = positions[neighbor] is_hl = key in hl_edges and not animate color = HIGHLIGHT_EDGE if is_hl else EDGE_COLOR width = 3 if is_hl else 1.2 cx, cy = (x1 + x2)//2, (y1 + y2)//2 - 15 canvas.create_line( x1, y1, cx, cy, x2, y2, smooth=True, splinesteps=24, fill=color, width=width ) mx, my = (x1 + x2)//2, (y1 + y2)//2 canvas.create_text(mx, my - 6, text=f"{attrs['distance']:.0f}", fill=TEXT_SECONDARY, font=('SF Pro', 7, 'bold')) # Nodes for name in nodes: x, y = positions[name] is_hl = highlight_path and name in highlight_path # Glow canvas.create_oval( x - RADIUS - 2, y - RADIUS - 2, x + RADIUS + 2, y + RADIUS + 2, fill='', outline=ACCENT_BLUE, width=2 ) # Node canvas.create_oval( x - RADIUS, y - RADIUS, x + RADIUS, y + RADIUS, fill=HIGHLIGHT_NODE if is_hl else NODE_FILL, outline=ACCENT_BLUE if is_hl else '#64748b', width=1.5 ) canvas.create_text(x, y, text=name[:3].upper(), fill=TEXT_PRIMARY, font=('SF Pro', 9, 'bold')) if animate and highlight_path: animate_path(canvas, highlight_path, positions) def animate_path(canvas, path, positions, i=0): if i >= len(path) - 1: return a, b = path[i], path[i+1] x1, y1 = positions[a] x2, y2 = positions[b] cx, cy = (x1 + x2)//2, (y1 + y2)//2 - 15 canvas.create_line( x1, y1, cx, cy, x2, y2, smooth=True, splinesteps=24, fill=HIGHLIGHT_EDGE, width=3 ) canvas.after(150, lambda: animate_path(canvas, path, positions, i+1)) # ───────────────────────────────────────── # SLICK COMPACT GUI # ───────────────────────────────────────── def build_gui(graph, nodes): root = tk.Tk() root.title("PathFinder") root.geometry(f"{WINDOW_W}x{WINDOW_H}") root.configure(bg=BG_COLOR) root.resizable(False, False) # ── Header ── header = tk.Frame(root, bg=BG_COLOR, height=60) header.pack(fill='x') header.pack_propagate(False) tk.Label(header, text="PATHFINDER", font=('SF Pro', 20, 'bold'), fg=ACCENT_BLUE, bg=BG_COLOR).pack(pady=12) # ── Compact Map ── canvas_frame = tk.Frame(root, bg=CARD_COLOR, relief='solid', bd=1) canvas_frame.pack(pady=12, padx=16, fill='x') canvas = tk.Canvas(canvas_frame, width=CANVAS_W, height=CANVAS_H, bg=BG_COLOR, highlightthickness=0) canvas.pack(pady=8) draw_map(canvas, graph, nodes) # ── Controls (Compact Grid) ── controls = tk.Frame(root, bg=BG_COLOR) controls.pack(pady=8, padx=16) # Style buttons style = ttk.Style() style.configure('Accent.TButton', font=('SF Pro', 9, 'bold')) tk.Label(controls, text="From", bg=BG_COLOR, fg=TEXT_SECONDARY, font=('SF Pro', 9, 'bold')).grid(row=0, column=0, sticky='w', pady=(0,4)) frm_var = tk.StringVar(value=nodes[0]) ttk.Combobox(controls, textvariable=frm_var, values=nodes, width=10, state='readonly', font=('SF Pro', 9)).grid(row=1, column=0, padx=(0,8)) tk.Label(controls, text="To", bg=BG_COLOR, fg=TEXT_SECONDARY, font=('SF Pro', 9, 'bold')).grid(row=0, column=1, sticky='w', pady=(0,4)) to_var = tk.StringVar(value=nodes[-1]) ttk.Combobox(controls, textvariable=to_var, values=nodes, width=10, state='readonly', font=('SF Pro', 9)).grid(row=1, column=1, padx=(0,8)) tk.Label(controls, text="Mode", bg=BG_COLOR, fg=TEXT_SECONDARY, font=('SF Pro', 9, 'bold')).grid(row=0, column=2, sticky='w', pady=(0,4)) opt_var = tk.StringVar(value='distance') ttk.Combobox(controls, textvariable=opt_var, values=['distance', 'time', 'fuel'], width=9, state='readonly', font=('SF Pro', 9)).grid(row=1, column=2) # ── Results Card ── result_card = tk.Frame(root, bg=CARD_COLOR, relief='solid', bd=1) result_card.pack(pady=12, padx=16, fill='x') result_var = tk.StringVar(value="Choose start, destination, and optimization mode.") result_label = tk.Label(result_card, textvariable=result_var, bg=CARD_COLOR, fg=TEXT_PRIMARY, font=('SF Pro', 10), justify='left', wraplength=560, pady=16) result_label.pack(padx=20) # ── Action Buttons ── btn_frame = tk.Frame(root, bg=BG_COLOR) btn_frame.pack(pady=(0,12), padx=16) def find_path(): start, end, key = frm_var.get(), to_var.get(), opt_var.get() if start == end: messagebox.showwarning("Error", "Start ≠ Destination") return cost, path, totals = dijkstra(graph, start, end, key) if not path: result_var.set(f"❌ No route: {start} → {end}") draw_map(canvas, graph, nodes) return path_str = " → ".join(path) result_var.set( f"✅ ROUTE: {path_str}\n\n" f"📏 Distance: {totals['distance']:.1f} km\n" f"⏱️ Time: {totals['time']:.0f} min\n" f"⛽ Fuel: {totals['fuel']:.1f} L" ) draw_map(canvas, graph, nodes, highlight_path=path, animate=True) def reset(): result_var.set("Choose start, destination, and optimization mode.") draw_map(canvas, graph, nodes) # Styled buttons find_btn = tk.Button(btn_frame, text="FIND ROUTE", command=find_path, bg=ACCENT_BLUE, fg='white', font=('SF Pro', 11, 'bold'), relief='flat', padx=24, pady=8, cursor='hand2') find_btn.grid(row=0, column=0, padx=(0,8)) reset_btn = tk.Button(btn_frame, text="RESET", command=reset, bg=CARD_COLOR, fg=TEXT_PRIMARY, font=('SF Pro', 10, 'bold'), relief='flat', padx=20, pady=8, cursor='hand2') reset_btn.grid(row=0, column=1) # Hover effects def on_enter(e, btn, bg): btn.config(bg=bg) def on_leave(e, btn, bg): btn.config(bg=bg) find_btn.bind("", lambda e: on_enter(e, find_btn, '#2563eb')) find_btn.bind("", lambda e: on_enter(e, find_btn, ACCENT_BLUE)) reset_btn.bind("", lambda e: on_enter(e, reset_btn, '#334155')) reset_btn.bind("", lambda e: on_enter(e, reset_btn, CARD_COLOR)) # ── Minimal Legend ── tk.Label(root, text="🔵 Nodes 🟠 Best Path ➤ Distance labels (km)", fg=TEXT_SECONDARY, bg=BG_COLOR, font=('SF Pro', 8)).pack() root.mainloop() # ───────────────────────────────────────── # ENTRY POINT # ───────────────────────────────────────── if __name__ == '__main__': CSV_FILE = os.path.join(os.path.dirname(__file__), 'Database.csv') try: graph, nodes = load_graph(CSV_FILE) except FileNotFoundError: print(f"ERROR: '{CSV_FILE}' not found.") raise print("Nodes:", nodes) print(f"Edges: {sum(len(v) for v in graph.values()) // 2}") build_gui(graph, nodes)
